CREATE TABLE [dbo].[Customer]
(
    [CustId] INT NOT NULL IDENTITY(1,1) PRIMARY KEY, 
    [ForeName] NVARCHAR(50) NULL, 
    [Surname] NVARCHAR(50) NULL, 
    [TeleNo] NCHAR(11) NULL, 
    [Address] NVARCHAR(50) NULL, 
    [Postcode] NCHAR(10) NULL
)

CREATE TABLE [dbo].[Staff]
(
    [StaffId] INT NOT NULL IDENTITY(1,1) PRIMARY KEY, 
    [Forename] VARCHAR(50) NULL, 
    [Surname] VARCHAR(50) NULL, 
    [TeleNo] NCHAR(11) NULL, 
    [Address] VARCHAR(50) NULL, 
    [Postcode] VARCHAR(7) NULL
)

CREATE TABLE [dbo].[Groups]
(
    [GroupId] INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
    [GroupName] varchar(20) NULL,
    [GroupAmount] INT NULL

)

CREATE TABLE [dbo].[GroupRota](
	[StaffID] INT FOREIGN KEY REFERENCES [dbo].[Staff]([StaffId]),
	[GroupID] INT FOREIGN KEY REFERENCES [dbo].[Groups]([GroupId])
	);

CREATE TABLE [dbo].[Kids]
(
        [ChildId] INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
	[ParentId] INT FOREIGN KEY REFERENCES [dbo].[Customer]([CustId]),
	[ForeName] varchar(50) NULL,
	[Surname] varchar(50) NULL,
	[DOB] VARCHAR(50) NULL
)

CREATE TABLE [dbo].[Booking]
(
        [BookingID] INT NOT NULL IDENTITY(1,1) PRIMARY KEY,
	[ChildId] INT FOREIGN KEY REFERENCES [dbo].[Kids]([ChildId]),
        [GroupId] INT FOREIGN KEY REFERENCES [dbo].[Groups]([GroupId]),
	[Monday] INT NULL,
    	[Tuesday]   INT NULL,
    	[Wednesday] INT NULL, 
    	[Thursday] INT NULL, 
    	[Friday] INT NULL, 
)

CREATE TABLE [dbo].[Dates]
(
	[BookingId] INT FOREIGN KEY REFERENCES [dbo].[Booking](BookingID), 
    	[Monday] VARCHAR(25) NOT NULL,
)

CREATE TABLE [dbo].[Prices]
(
    [Base] INT NOT NULL, 
    [MinDiscount] INT NULL, 
    [MedDiscount] INT NULL, 
    [MaxDiscount] INT NULL 
)
